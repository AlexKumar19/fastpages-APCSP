function Palindrome() {
    console.log(document.getElementById("input").value)
    const word = document.getElementById("input").value
    var reverse = ""
    for (var i = word.length - 1; i >= 0; i--) { 
        reverse += word[i]; 
    }
    console.log(reverse)
    if (reverse== word) {
        document.getElementById("text").value = "this is a palindrome"
