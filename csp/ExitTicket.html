    <script>
        // JavaScript logic

        // Function to shuffle an array using Fisher-Yates algorithm
        function shuffleArray(array) {
            for (let i = array.length - 1; i > 0; i--) {
                let j = Math.floor(Math.random() * (i + 1));
                [array[i], array[j]] = [array[j], array[i]];
            }
            return array;
        }

        // Function to animate the shuffling of boxes
        function animateShuffle() {
            const boxes = Array.from(document.getElementsByClassName('box'));
            const shuffledBoxes = shuffleArray(boxes);

            shuffledBoxes.forEach((box, index) => {
                setTimeout(() => {
                    box.style.order = index + 1;
                }, (index + 1) * 500);
            });
        }

        // Function to animate the sorting of boxes using bubble sort algorithm
        function animateSort() {
            const boxes = Array.from(document.getElementsByClassName('box'));
            let sorted = false;
            let pass = 0;

            function swap(a, b) {
                const temp = boxes[a].style.order;
                boxes[a].style.order = boxes[b].style.order;
                boxes[b].style.order = temp;
            }

            function bubbleSort() {
                sorted = true;
                for (let i = 0; i < boxes.length - pass - 1; i++) {
                    if (parseInt(boxes[i].style.order) > parseInt(boxes[i + 1].style.order)) {
                        swap(i, i + 1);
                        sorted = false;
                    }
                }
                pass++;

                if (!sorted) {
                    setTimeout(bubbleSort, 500);
                }
            }

            bubbleSort();
        }

        // Triggering the animations
        animateShuffle();
        setTimeout(animateSort, 5000);
    </script>
