import random
import math
def generator(range):
    return math.floor(random.random()*range)
number = generator(8)
if number <= 3:
    print("green")
elif number > 3 and number <6:
    print("blue")
elif number ==6:
    print("purple")
elif number==7:
    print("orange")
elif number==8:
    print("red")
